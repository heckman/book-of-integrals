\begin{gathered}[t]\textbf{Half-Angle Formulae}\\
\begin{aligned}

\end{aligned}\end{gathered}

\begin{gathered}[t]\textbf{Double-Angle Formulae}\\
\begin{aligned}
  \cos{2x}&=\cos^2x-\sin^2x\\&=2\cos^2x-1\\&=1-2\sin^2x\\[0.5em]
  \sin{2x}&=2\sin x \cos x\\[0.5em]

\end{aligned}\end{gathered}
