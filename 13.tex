\def\theintegral{\(\int\cot{u}\dif{u}=\ln\abs{\sin{u}} + C\)}

\begin{align*}
% \renewcommand{\baselinestretch}{2}
I_{13}
=& \int  \cot{u} \dif{u} \\
=& \int  \frac{\cos{u}}{\sin{u}} \dif{u}
\nb\sub[\quad  v=\sin{u} \andso \dif{v} =& \cos{u}\dif{u}  ]
=&  \int \frac{1}{v}\dif{v} \\
=&  \ln \abs{v} + C
\nb\unsub
=&  \ln \abs{\sin{u}} + C
\end{align*}
